command for checking git version:
   git version

command for listing config details:
   git config --list

command for adding username:
   git config --global user.email "example@gmail.com"

command for initialising git respository:  
   git init

command for adding all untracked files to staging area:  
   git add -A  or   git add .
   
command for adding staged files to local repository:  
    git commit -m "commit message"

command for listing all untracked files:
    git status

command for adding remote repository to local repository:
   git remote add name remote_path

command for pushing local repository to remote repository:
   git push name branch

command for getting branch name:
   git branch
